CREATE TABLE `region` (
  `id_region` int(11) NOT NULL AUTO_INCREMENT,
  `nombre` varchar(45) NOT NULL,
  PRIMARY KEY (`id_region`)
) 

CREATE TABLE `votaciones`.`comunas` (
  `id_comuna` INT NOT NULL AUTO_INCREMENT,
  `id_region` INT NULL,
  `nombre` VARCHAR(45) NULL,
  PRIMARY KEY (`id_comuna`),
  INDEX `fk_region_comuna_idx` (`id_region` ASC),
  CONSTRAINT `fk_region_comuna`
    FOREIGN KEY (`id_region`)
    REFERENCES `votaciones`.`region` (`id_region`)
    ON DELETE NO ACTION
    ON UPDATE CASCADE);


CREATE TABLE `votaciones`.`candidatos` (
  `id_candidatos` INT NOT NULL,
  `nombre` VARCHAR(45) NULL);

CREATE TABLE `votaciones`.`votos` (
  `idvotos` INT NOT NULL AUTO_INCREMENT,
  `nombre` VARCHAR(45) NULL,
  `alias` VARCHAR(45) NULL,
  `rut` VARCHAR(45) NULL,
  `email` VARCHAR(45) NULL,
  `region` VARCHAR(45) NULL,
  `comuna` VARCHAR(45) NULL,
  `candidato` VARCHAR(45) NULL,
  PRIMARY KEY (`idvotos`),
  UNIQUE INDEX `rut_UNIQUE` (`rut` ASC) );
